179  // socket is configured as a streaming TCP/IP server.
180  struct addrinfo hints;
181  memset(&hints, 0, sizeof(struct addrinfo));
182  hints.ai_flags = traits::flags(); // Socket will be passed to bind().
183  hints.ai_family = traits::family(); // Internet Protocol (IP) socket.
184  hints.ai_socktype = traits::type(); // Streaming socket.
185  hints.ai_protocol = traits::protocol(); // TCP protocol
186 
187  // Retrieves the socket address for the host. Here "localhost" is used so that the server
188  // is available only on the current system. A real server would be accessible to the
189  // outside world so the hostname would be either one that can be resolved to an external
190  // IP, or simply NULL to be accessible on all attached networks. The second parameter is
191  // the service (port number) that the server will listen on.
192  int err = 0;
193  if((err = ::getaddrinfo(name.c_str(), std::to_string(port).c_str(), &hints, &res0_))) {
194  std::ostringstream msg;

196  throw net::socket_error(msg.str());
197  }
